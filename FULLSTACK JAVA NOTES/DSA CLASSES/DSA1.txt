Q) Write a sample to sort data (numbers) in an 2-dimensional array[4*5]. using two loops only?


class Test  
{
	public static void main(String[] args) 
	{
		int[][] arr = {
						{6,1,9},
						{2,4,5},
						{3,7,8}
					};
		
		int rows = arr.length;
		int cols = arr[0].length;
		int total = rows * cols;
		
		int[] flat = new int[total];
		
		//copy the data from 2-D to 1-D 
		int index=0;
		for(int i=0;i<rows;i++)
		{
			for(int j=0;j<cols;j++)
			{
				flat[index++]=arr[i][j];
			}
		}
		
		
		//sorting logic 
		for(int i=0;i<flat.length;i++)
		{
			for(int j=i+1;j<flat.length;j++)
			{
				if(flat[i]>flat[j])
				{
					//swapping 
					int temp = flat[i];
					flat[i]=flat[j];
					flat[j]=temp;
				}
			}
		}
		
	
		//convert 1-D to 2-D 
		int idx=0;
		for(int i=0;i<rows;i++)
		{
			for(int j=0;j<cols;j++)
			{
					arr[i][j]= flat[idx++];
			}
		}
		
		
		//display the data 
		for(int i=0;i<rows;i++)
		{
			for(int j=0;j<cols;j++)
			{
				System.out.print(arr[i][j]+" ");
			}
			//new line 
			System.out.println();
		}
		
	}
}

Q) Write a program using lamda expression to implements functional interface (without custom interface) to perform arithmetic operations?


BiFunction is a functional interface, which accepts two arguments and returns a result. The interface contains two methods: The apply() method. The andThen() method.

ex:
---
import java.util.function.*;
class Test  
{
	public static void main(String[] args) 
	{
		BiFunction<Integer,Integer,Integer> add = (i1,i2) ->  i1+i2;
		BiFunction<Integer,Integer,Integer> subtract = (i1,i2) -> i1-i2;
		
		int a=10,b=20;
		System.out.println("Addition : "+add.apply(a,b));
		System.out.println("Subtraction : "+subtract.apply(a,b));
	}
}

Q) Given a list of strings , find out all the string starting with vowels using stream functions?

input:
	apple, banana, orange, umbrella, grape, elephant 


import java.util.*;
import java.util.stream.*;
class Test  
{
	public static void main(String[] args) 
	{
		List<String> list = List.of("apple","banana","orange","umbrella","grape","elephant");
		
		list.stream().filter(s -> { 
			char ch = Character.toLowerCase(s.charAt(0));
			return ch=='a' || ch=='e'|| ch=='i' || ch=='o' || ch=='u';
			}).forEach(s -> System.out.print(s+" "));
			
		
	}
}

Q) Write a java program to reverse an array without using another array?

input:
	1 2 3 4 5

output:
	5 4 3 2 1 

class Test  
{
	public static void main(String[] args) 
	{
		int[] arr={1,2,3,4,5};		
		
		int start=0;
		int end=arr.length-1;
		
		while(start<end)
		{
			int temp = arr[start];
			arr[start]=arr[end];
			arr[end]=temp;
			
			start++;
			end--;
		}
		
		//display elements
		for(int i:arr)
		{
			System.out.print(i+" ");
		}
		
	}
}

Q) Merge two sorted list into a single sorted list using java streams?

input:
	1 3 5 7 9
	2 4 6 8 10


import java.util.*;
import java.util.stream.*;
class Test  
{
	public static void main(String[] args) 
	{
		List<Integer> list1 = List.of(1,3,5,7,9);
		List<Integer> list2 = List.of(2,4,6,8,10);
		
		List<Integer> newList = Stream.concat(list1.stream(),list2.stream())
			.sorted().collect(Collectors.toList());
		
		newList.forEach(e -> System.out.print(e+" "));

	}
}
